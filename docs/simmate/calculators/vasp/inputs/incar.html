 18class Incar(dict):
 19    """
 20    INCAR object for reading and writing INCAR files. This behaves exactly like
 21    a python dictionary, but has a few extra checks and methods attached to it.
 22    You can pass it a dictionary or initialize it just like you would dict(kwargs).
 23    You can consider the dict(kwargs) as equivalent to Incar(parameters).
 24
 25    If you want a given setting to be dependent on the structure or dynamically
 26    determined, then we implement these modifiers. This would enable us to
 27    do things like ENCUT__per_atom or NGZF__density. We can even have more complex
 28    modifiers like multiple_keywords__smart_ldau which signals that our
 29    "smart_ldau" modifier introduces more than one new setting to the INCAR, such
 30    as LDAUJ, LDAUU, LDAUL, LDAUTYPE, and LDAUPRINT.
 31    """
 32
 33    def __init__(self, **kwargs):
 34
 35        # The kwargs are a dictionary of parameters (e.g. {"PREC": "accurate"})
 36
 37        # Establish the dictionary with the given input
 38        # This is the same as doing dict(**kwargs)
 39        super().__init__(**kwargs)
 40
 41        # Based on the kwargs, go through and make sure each parameter is in
 42        # it's expected datatype. Thus users can supply strings in the typical
 43        # VASP format that will be convert to python types here.
 44        # OPTIMIZE -- would it be faster if I only did this on from_file init?
 45        # and therefore assumed the user to initialize this with proper python
 46        # datatypes and formatting? Also will this behave properly if the value
 47        # is already in the correct format?
 48        for parameter, value in self.items():
 49
 50            # parameters may have tags like "__density" added onto them. We don't
 51            # convert to the datatype yet, but instead wait until a structure
 52            # is provided (when the to_file or str methods are called below).
 53            # These are defined in python and should already be in the correct
 54            # data format.
 55            if "__" in parameter:
 56                self.update({parameter: value})
 57
 58            # Otherwise, we need to convert values to the proper python datatype
 59            # because we might be reading from a file where everything is a string
 60            else:
 61                formatted_value = self._str_to_datatype(parameter, value)
 62                self.update({parameter: formatted_value})
 63
 64        # SPECIAL CASE
 65        # If you have LSORBIT=True or LNONCOLLINEAR=True, the MAGMOM class needs
 66        # to take on a different form. Instead of a single MAGMOM being set as
 67        # a float, it's now a vector (x,y,z). For example:
 68        #   MAGMOM = x1 y1 z1    x2 y2 z2    x3 y3 z3 ...
 69        #       [...instead of...]
 70        #   MAGMOM = value1 value2 value3 ...
 71        # Check if MAGMOM is set along with LSORBIT or LNONCOLLINEAR being True
 72        if ("MAGMOM" in kwargs) and (
 73            kwargs.get("LSORBIT") or kwargs.get("LNONCOLLINEAR")
 74        ):
 75            # convert the MAGMOM value from...
 76            #   [x1,y1,z1,x2,y2,z2,x3,y3,z3] (this can be a string too)
 77            #   to...
 78            #   [[x1,y1,z1],[x2,y2,z2],[x3,y3,z3]]
 79            old_format = kwargs["MAGMOM"]
 80            new_format = [old_format[i : i + 3] for i in range(0, len(old_format), 3)]
 81            # Or If I wanted to do the same thing with numpy:
 82            #   new_format = numpy.array_split(old_format, len(old_format) // 3)
 83            # now update the dictionary with this value
 84            self.update({"MAGMOM": new_format})

 86    def to_evaluated_str(self, structure=None):
 87
 88        # Let's start with an empty string and build from there
 89        final_str = ""
 90
 91        # First we need to iterate through all parameters and check if we have
 92        # ones that are structure-specific. For example, we would need to
 93        # evaluate "ENCUT__per_atom". We go through all these and collect the
 94        # parameters into a final settings list.
 95        final_settings = {}
 96        for parameter, value in self.items():
 97
 98            # if there is no modifier attached to the parameter, we just keep it as-is
 99            if "__" not in parameter:
100                final_settings[parameter] = value
101
102            # Otherwise we have a modifier like "__density" and need to evaluate it
103            else:
104
105                # make sure we have a structure supplied because all modifiers
106                # require one.
107                if not structure:
108                    raise Exception(
109                        "It looks like you used a keyword modifier but didn't "
110                        f"supply a structure! If you want to use {parameter}, "
111                        "then you need to make sure you provide a structure so "
112                        "that the modifier can be evaluated."
113                    )
114
115                # separate the input into the base parameter and modifier.
116                # This also overwrites what our paramter value is.
117                parameter, modifier_tag = parameter.split("__")
118
119                # check that this class has this modifier supported. It should
120                # be a method named "keyword_modifier_mymodifier".
121                # If everything looks good, we grab the modifier function
122                modifier_fxn_name = "keyword_modifier_" + modifier_tag
123                if hasattr(self, modifier_fxn_name):
124                    modifier_fxn = getattr(self, modifier_fxn_name)
125                else:
126                    raise AttributeError(
127                        """
128                        It looks like you used a keyword modifier that hasn't
129                        been defined yet! If you want something like ENCUT__smart_encut,
130                        then you need to make sure there is a keyword_modifier_smart_encut
131                        method available.
132                        """
133                    )
134
135                # now that we have the modifier function, let's use it to update
136                # our value for this keyword.
137                value = modifier_fxn(structure, value)
138
139                # sometimes the modifier returns None. In this case we don't
140                # set anything in the INCAR, but leave it to the programs
141                # default value.
142                if not value:
143                    continue
144                # BUG: Are there cases where None is return but we still want
145                # to write it to the INCAR? If so, it'd be skipped here.
146
147                # if the "parameter" is actually "multiple_keywords", then we
148                # have our actual parameters as a dictionary. We need to
149                # pull these out of the "value" we have.
150                if parameter == "multiple_keywords":
151                    for subparameter, subvalue in value.items():
152                        final_settings[subparameter] = subvalue
153
154                # otherwise we were just given back an update value
155                else:
156                    final_settings[parameter] = value
157
158        # Now that we have all of our parameters evaluated for the structure, we
159        # iterate through each parameter and its set value. Each one will be
160        # put on a separate line.
161        for parameter, value in final_settings.items():
162
163            # let's start by adding the parameter key to our output
164            # It will be followed by an equal sign to separate it's value
165            final_str += parameter + " = "
166
167            # If we have a value that is a list (or list of lists)
168            if isinstance(value, list):
169
170                # check if we have a list of lists
171                # Take MAGMOM with LSORBIT=True as an example, where we convert...
172                #   [[x1,y1,z1],[x2,y2,z2],[x3,y3,z3]]
173                # to...
174                # "x1 y1 z1 x2 y2 z2 x3 y3 z3"
175                if isinstance(value[0], list):
176                    final_str += " ".join(str(n) for xyz in value for n in xyz)
177
178                # otherwise we just have a list of items
179                # converts [1, 2, 3] to "1 2 3"
180                else:
181                    final_str += " ".join(str(n) for n in value)
182
183            # if it's not a list, we can just save the string of the value
184            else:
185                final_str += str(value)
186
187            # we want to start each parameter on a new line, so we need to
188            # add this at the end of each. There will be an extra new line at
189            # the end of the file, but that's alright
190            final_str += "\n"
191
192        # we now have our final string and can return it!
193        return final_str

195    def to_file(self, filename="INCAR", structure=None):
196        """
197        Write Incar to a file.
198        Args:
199            filename (str): filename to write to.
200        """
201        # we just take the string format and put it in a file
202        with open(filename, "w") as file:
203            file.write(self.to_evaluated_str(structure=structure))

205    @staticmethod
206    def from_file(filename="INCAR"):
207        """
208        Reads an Incar object from a file.
209        Args:
210            filename (str): Filename for file
211        Returns:
212            Incar object
213        """
214        # open the file, grab the lines, and then close it
215        with open(filename) as file:
216            lines = file.readlines()
217
218        # store parameters in this dictionary
219        parameters = {}
220        # go through line by line
221        for line in lines:
222            # If the line starts with a # then its a comment and we should skip.
223            # It also could be an empty line that we should skip,
224            if line.startswith("#") or line.startswith("\n"):
225                continue
226            # multiple parameters on a single line are separated by a semicolon
227            for sub_line in line.split(";"):
228                # the PARAMETER and VALUE are separated by equal sign
229                parameter, value = sub_line.split("=")
230                # we can use the parameter/value to update our dictionary. The
231                # last thing we do is remove leading/trailing whitespace with strip()
232                parameters[parameter.strip()] = value.strip()
233
234        # return the final dictionary as an Incar object
235        return Incar(**parameters)

239        """
240        When given a vasp parameter and it's value as a string, this helper
241        function will use the key (parameter) to determine how to convert the
242        val string to the proper python datatype (int, float, bool, list...).
243        I have the most common keys mapped out, but if a parameter is given that
244        isn't mapped, I simply leave it as a string.
245        """
246
247        # I outline the most common keys to what their expected data types are.
248
249        # OPTIMIZE -- I should set these elsewhere so that these lists are not
250        # initialized every time I call this function. Maybe have a dictionary
251        # of {Parameter: Value_datatype} in the main enviornment for use.
252
253        vector_list_keys = (
254            # "MAGMOM",  # depends on other args -- see notes in init
255            "DIPOL",
256        )
257
258        float_list_keys = (
259            "LDAUU",
260            "LDAUJ",
261            "MAGMOM",  # depends on other args -- see notes in init
262            "LANGEVIN_GAMMA",
263            "QUAD_EFG",
264            "EINT",
265        )
266        int_list_keys = (
267            "LDAUL",
268            "LDAUJ",
269            "EINT",
270        )
271
272        bool_keys = (
273            "LDAU",
274            "LWAVE",
275            "LSCALU",
276            "LCHARG",
277            "LPLANE",
278            "LUSE_VDW",
279            "LHFCALC",
280            "ADDGRID",
281            "LSORBIT",
282            "LNONCOLLINEAR",
283            "KGAMMA",
284        )
285        float_keys = (
286            "EDIFF",
287            "SIGMA",
288            "TIME",
289            "ENCUTFOCK",
290            "HFSCREEN",
291            "POTIM",
292            "EDIFFG",
293            "AGGAC",
294            "PARAM1",
295            "PARAM2",
296            "KSPACING",
297            "SYMPREC",
298            "AMIX",
299            "BMIX",
300            "AMIN",
301            "SMASS",
302            "AMIX_MAG",
303            "BMIX_MAG",
304        )
305        int_keys = (
306            "NSW",
307            "NBANDS",
308            "NELMIN",
309            "ISIF",
310            "IBRION",
311            "ISPIN",
312            "ICHARG",
313            "NELM",
314            "ISMEAR",
315            "NPAR",
316            "LDAUPRINT",
317            "LMAXMIX",
318            "ENCUT",
319            "NSIM",
320            "NKRED",
321            "NUPDOWN",
322            "ISPIND",
323            "LDAUTYPE",
324            "IVDW",
325            "ISTART",
326            "NELMDL",
327            "IMIX",
328            "ISYM",
329        )
330
331        # If the value is not a string, then assume we are already in the
332        # correct format. Note, an incorrect format will throw an error
333        # somewhere below, which may be tricky for beginners to traceback.
334        if not isinstance(value, str):
335            return value
336
337        # if the parameter is in int_keys
338        if parameter in int_keys:
339            # sometimes "1." was written to indicate an integer so check for
340            # this and remove it if needed.
341            if value[-1] == ".":
342                value = value[:-1]
343            # return the value integer
344            return int(value)
345
346        # if the parameter is in float_keys, we convert value to a float
347        elif parameter in float_keys:
348            # return the value float
349            return float(value)
350
351        # if the parameter is in bool_keys
352        elif parameter in bool_keys:
353            # Python is weird where bool("FALSE") will return True... So I need
354            # to convert the string to lowercase and read it to know what to
355            # return here.
356            if "t" in value.lower():
357                return True
358            elif "f" in value.lower():
359                return False
360
361        # if the parameter is in vector_list_keys
362        # These vectors are always floats
363        elif parameter in vector_list_keys:
364            # convert a string of...
365            #   "x1 y1 z1 x2 y2 z2 x3 y3 z3"
366            # to...
367            #   [x1,y1,z1,x2,y2,z2,x3,y3,z3] (list of floats)
368            # and then to...
369            #   [[x1,y1,z1],[x2,y2,z2],[x3,y3,z3]]
370            value = [float(item) for item in value.split()]
371            return [value[i : i + 3] for i in range(0, len(value), 3)]
372
373        # if the parameter is in float_list_keys
374        elif parameter in float_list_keys:
375            final_list = []
376            for item in value.split():
377                # Sometimes, the values are given as "3*0.1 2*0.5" where the "*"
378                # means to include that value that many times. For example, this
379                # input would be the same as "0.1 0.1 0.1 0.5 0.5". We need to
380                # account for this when parsing.
381                if "*" in item:
382                    nsubitems, subitem = item.split("*")
383                    for n in range(int(nsubitems)):
384                        final_list.append(float(subitem))
385                else:
386                    final_list.append(float(item))
387            return final_list
388
389        # if the parameter is in int_list_keys
390        elif parameter in int_list_keys:
391            return [int(item) for item in value.split()]
392
393        # If it is not in the common keys listed, just leave it as a string.
394        else:
395            return value

397    def compare_incars(self, other_incar):
398        """
399        Compares two Incars and indicates which parameters are the same and
400        which are not. Useful for checking whether two runs were done using
401        the same parameters.
402        Args:
403            other (Incar): The other Incar object to compare to.
404        Returns:
405            Dict of the following format:
406            {"Same" : parameters_that_are_the_same,
407            "Different": parameters_that_are_different}
408            The value of the other_incar is returned for the paramters that
409            are different are returned as {Parameter: (incar_value, other_incar_value)}
410            where None is used as a placeholder. For the same parameters, the
411            dictionary is returned as {Parameter: shared_value}
412        """
413
414        same_parameters = {}
415        different_parameters = {}
416
417        # make a collection of all unique keys used by the two dictionaries
418        # I convert to list() so I can use set() below.
419        # OPTIMIZE -- is there a better way to do this?
420        parameters1 = list(self.keys())
421        parameters2 = list(other_incar.keys())
422
423        # iterate through each unique parameter
424        for parameter in set(parameters1 + parameters2):
425
426            # try to grab the value from both Incar objects
427            # If it doesn't have the value, None will be provided
428            value1 = self.get(parameter)
429            value2 = other_incar.get(parameter)
430
431            # compare the two and store
432            if value1 == value2:
433                # it doesn't matter which value I grab so I just use value1
434                same_parameters[parameter] = value1
435            # if they are different...
436            else:
437                different_parameters[parameter] = (value1, value2)
438
439        return {"Same": same_parameters, "Different": different_parameters}

441    def __add__(self, other):
442        """
443        Add all the values of another INCAR object to this object.
444        Facilitates the use of "standard" INCARs.
445        """
446        # start by copying this incar's settings
447        new_parameters = dict(self.items())
448
449        # now iterate through the other incar parameters to incorporate them
450        for parameter, value in other.items():
451
452            # If both have a given parameter, make sure they are the same values
453            if parameter in self and value != self[parameter]:
454                # If not, we shouldn't allow combining these Incars as it can
455                # lead to undesired results.
456                raise ValueError(
457                    f"Incars have conflicting values! One conflict is with {parameter} "
458                    f"where one incar has {self[parameter]} and the other has {value}"
459                )
460            # otherwise just set the value
461            else:
462                new_parameters[parameter] = value
463
464        # now return the new Incar
465        return Incar(**new_parameters)

467    @classmethod
468    def add_keyword_modifier(cls, keyword_modifier: Callable):
469        """
470        Dynamically sets a keyword_modifier function to the Incar class. When
471        loaded onto the Incar class, you can then use the modifiers name to
472        set a parameter based on the input structure.
473
474        Names of modifiers MUST start with "keyword_modifier_" and
475        have unique endings (i.e. you can't have 2 "per_atom" modifiers).
476
477        For example, in order to allow settings like `ENCUT__per_atom`, we use
478        a `keyword_modifier_per_atom` function and add it to this class:
479
480        ``` python
481
482        # create your modifier logic as a function
483        def keyword_modifier_double_per_atom(structure, per_atom_value):
484            return per_atom_value * structure.num_sites * 2
485
486        # add it to the Incar class to make it active
487        Incar.add_keyword_modifier(keyword_modifier_double_per_atom)
488
489        # You can not use the modifier in incar logic!
490        incar = Incar(ENCUT__double_per_atom=1e-5)
491        ```
492        """
493
494        # first ensure that the modifier does not conflict with an existing one
495        if hasattr(cls, keyword_modifier.__name__):
496            raise Exception(
497                "The Incar class already has a modifier with this name. "
498                "Please use a different modifier name to avoid conflicts."
499            )
500
501        # add the function/callable as a static method to this class.
502        # This line of code is based on...
503        #   https://stackoverflow.com/questions/17929543/
504        setattr(cls, keyword_modifier.__name__, staticmethod(keyword_modifier))
505
506
507# set some default keyword modifiers
508for modifier in [
509    keyword_modifier_density,
510    keyword_modifier_density_a,
511    keyword_modifier_density_b,
512    keyword_modifier_density_c,
513    keyword_modifier_per_atom,
514    keyword_modifier_smart_magmom,
515    keyword_modifier_smart_lmaxmix,
516    keyword_modifier_smart_ldau,
517    keyword_modifier_smart_ismear,
518]:
519    Incar.add_keyword_modifier(modifier)

123                modifier_fxn_name = "keyword_modifier_" + modifier_tag

399        """
400        Compares two Incars and indicates which parameters are the same and
401        which are not. Useful for checking whether two runs were done using
402        the same parameters.
403        Args:
404            other (Incar): The other Incar object to compare to.
405        Returns:
406            Dict of the following format:
407            {"Same" : parameters_that_are_the_same,
408            "Different": parameters_that_are_different}
409            The value of the other_incar is returned for the paramters that
410            are different are returned as {Parameter: (incar_value, other_incar_value)}

412            dictionary is returned as {Parameter: shared_value}
413        """
414
415        same_parameters = {}
416        different_parameters = {}
417
418        # make a collection of all unique keys used by the two dictionaries
419        # I convert to list() so I can use set() below.
420        # OPTIMIZE -- is there a better way to do this?
421        parameters1 = list(self.keys())
422        parameters2 = list(other_incar.keys())
423
424        # iterate through each unique parameter
425        for parameter in set(parameters1 + parameters2):
426
427            # try to grab the value from both Incar objects
428            # If it doesn't have the value, None will be provided
429            value1 = self.get(parameter)
430            value2 = other_incar.get(parameter)
431
432            # compare the two and store
433            if value1 == value2:
434                # it doesn't matter which value I grab so I just use value1
435                same_parameters[parameter] = value1
436            # if they are different...
437            else:
438                different_parameters[parameter] = (value1, value2)
439
440        return {"Same": same_parameters, "Different": different_parameters}

445        Facilitates the use of "standard" INCARs.
446        """

448        new_parameters = dict(self.items())
449

58def keyword_modifier_smart_magmom(structure: Structure, override_values: dict):
59    """
60    The __smart_magmom modifier goes through a series of checks to decide
61    what to set the MAGMOM as for VASP. In order of priority they are...
62        (1) the magmom property attached to each site in the structure object
63        (2) the spin property attached to the site's specie
64        (3) a value provided explicitly (e.g. {"Co": 0.5})
65        (4) a value of 0.6
66    """
67
68    # grab the default MAGMOM supplied, or use VASP's default of 1 otherwise
69    default_value = override_values.get("default", 1)
70
71    # we go through each site in the structure and decide what to set the
72    # MAGMOM for each. This allows even different sites of the same
73    # element to have their own MAGMOM
74    magnetic_moments = []
75    for site in structure:
76        # if the structure object has magmom-decorated sites, we use that
77        # as our first priority
78        if hasattr(site, "magmom"):
79            magnetic_moments.append(site.magmom)
80        # next we check if the site's specie has a spin
81        elif hasattr(site.specie, "spin"):
82            magnetic_moments.append(site.specie.spin)
83        # we then look at the override dictionary if there was one provided.
84        # If note, we use 0.6 as a default.
85        else:
86            magnetic_moment = override_values.get(site.specie.symbol, default_value)
87            magnetic_moments.append(magnetic_moment)
88
89    # This feature is in pymatgen, but I haven't added it here yet.
90    # if self.constrain_total_magmom:
91    #     nupdown = sum([mag if abs(mag) > 0.6 else 0 for mag in incar["MAGMOM"]])
92    #     incar["NUPDOWN"] = nupdown
93
94    return magnetic_moments

def keyword_modifier_smart_ldau( structure: simmate.toolkit.base_data_types.structure.Structure, ldau_config: dict) 
123def keyword_modifier_smart_ldau(structure: Structure, ldau_config: dict):
124    """
125    This modifier handles a series of keyword arguments that are associated
126    with LDAU, including LDAUJ, LDAUL, LDAUTYPE, LDAUU, and LDAUPRINT.
127    Therefore, a complex dictionary is passed to this. The format looks
128    like this...
129        LDAU__multiple_keywords__smart_ldau = dict(
130            LDAU__auto=True,
131            LDAUTYPE=2,
132            LDAUPRINT=1,
133            LDAUJ={...},
134            LDAUL={...},
135            LDAUU={...},
136            )
137    The LDAUJ, LDAUL, and LDAUU values can be a dictionary of elements to
138    value, or (most commonly) a nested dictionary. For example...
139        LDAUJ = {"F":{"Co":0}}
140    This would mean if the structure is a fluoride, set the LDAUJ for Co to 0.
141    If there are multiple options here (e.g. for fluorides and oxides), then
142    priority is placed on the most electronegative element. This would mean
143    some thing like yttrium oxide fluoride would be treated as a fluoride over
144    an oxide.
145    """
146
147    # first we need to go through the LDAUJ, LDAUL, and LDAUU keywords and
148    # see what their values are. If all of these end up be 0 for all elements
149    # then we actually don't need LDAU at all! Therefore, we'll go through
150    # all of these keywords and build a dictionary of settings to return
151    # at the end of this function
152    ldau_settings = {}
153
154    # To help decide how we set these values, let's check what the most
155    # electronegative element is, which will be last in the sorted composition
156    most_electroneg = sorted(structure.composition.elements)[-1].symbol
157
158    # As we go through these settings, we want see if we are even using LDAU.
159    # For example if we ran a calculation on NaCl, we'd probably see that
160    # all LDAUJ/LDAUL/LDAUU values are just 0. In that case, we can just
161    # throw away (i.e. turn off) all LDAU settings. Therefore, we have a term
162    # "using_ldau" that is false until proven otherwise.
163    using_ldau = False
164
165    for ldau_keyword in ["LDAUJ", "LDAUL", "LDAUU"]:
166
167        # grab the sub-dictionary that maps elements to this keyword.
168        # If it's not there, just use an empty dictionary.
169        keyword_config = ldau_config.get(ldau_keyword, {})
170
171        # check if the most electronegative element is in the override_values
172        # and if so, see if it has a subdictionary in it. We use this
173        # dictionary as our base one to pull values from.
174        if most_electroneg in keyword_config and isinstance(
175            keyword_config[most_electroneg], dict
176        ):
177            keyword_config = keyword_config[most_electroneg]
178
179        # now iterate through all the sites and grab the assigned value. If nothing
180        # is set then the default is 0.
181        values = [
182            keyword_config.get(element.symbol, 0) for element in structure.composition
183        ]
184
185        # now that we have this keyword all set, we add it to our results
186        ldau_settings[ldau_keyword] = values
187
188        # check to see if we are using ldau here (i.e. any value is not 0)
189        if any(values):
190            using_ldau = True
191
192    # now check if we actaully need LDAU here. If not, we can throw out all
193    # settings and just return an empty dictionary
194    if using_ldau and "LDAU__auto" in ldau_config:
195        ldau_settings["LDAU"] = True
196    else:
197        return {}
198
199    # The remaining LDAU keywords are LDAUPRINT and LDAUTYPE, which we just
200    # leave at what is set in the input
201    if "LDAUPRINT" in ldau_config:
202        ldau_settings["LDAUPRINT"] = ldau_config["LDAUPRINT"]
203    if "LDAUTYPE" in ldau_config:
204        ldau_settings["LDAUTYPE"] = ldau_config["LDAUTYPE"]
205
206    return ldau_settings
