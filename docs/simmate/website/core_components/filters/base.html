 85        ):
 86            class Meta:
 87                model = MatProjStructure  # this is database table
 88                fields = {...} # this combines the fields from Structure/Thermo mixins
 89
 90            # These attributes are set using the declared filters from Structure/Thermo mixins
 91            declared_filter1 = ...
 92            declared_filter1 = ...
 93        ```
 94        """
 95
 96        # this must be imported locally because it depends on all other classes
 97        # from this module -- and will create circular import issues if outside
 98        from simmate.website.core_components import filters as simmate_filters
 99
100        # First we need to grab the parent mixin classes of the table. For example,
101        # the MatProjStructure uses the database mixins ['Structure', 'Thermodynamics']
102        # while MatProjStaticEnergy uses ["StaticEnergy"].
103        mixin_names = [base.__name__ for base in table.__bases__]
104
105        # Because our Forms follow the same naming conventions as
106        # simmate.database.base_data_types, we can simply use these mixin names to
107        # load a Form mixin from the simmate.website.workflows.form module. We add
108        # these mixins onto the standard ModelForm class from django.
109        filter_mixins = [cls]
110        filter_mixins += [
111            getattr(simmate_filters, name)
112            for name in mixin_names
113            if hasattr(simmate_filters, name)
114        ]
115
116        # combine the fields of each filter mixin. Note we use .get_fields instead
117        # of .fields to ensure we get a dictionary back
118        filter_fields = {
119            field: conditions
120            for mixin in filter_mixins
121            for field, conditions in mixin.get_fields().items()
122        }
123
124        # Also combine all declared filters from each mixin
125        filters_declared = {
126            name: filter_obj
127            for mixin in filter_mixins
128            for name, filter_obj in mixin.declared_filters.items()
129        }
130
131        # The FilterSet class requires that we set extra fields in the Meta class.
132        # We define those here. Note, we accept all fields by default and the
133        # excluded fields are only those defined by the supported form_mixins -
134        # and we skip the first mixin (which is always DatabaseTableFilter)
135        class Meta:
136            model = table
137            fields = filter_fields
138
139        extra_attributes = {"Meta": Meta, **filters_declared}
140        # BUG: __module__ may need to be set in the future, but we never import
141        # these forms elsewhere, so there's no need to set it now.
142
143        # Now we dynamically create a new form class that we can return.
144        NewClass = type(table.__name__, tuple(filter_mixins), extra_attributes)
145
146        return NewClass

423            new_field2 = table_column.FloatField()
424            # app_label --> typically not required bc the parent class sets this
425        )
426        ```

429        code when we need to redefine a bunch of relationships in every single
430        child class (and always in the same way). A great example of it's utility

432        """

470        each of the `_from_toolkit` methods of those mixins and combines the results.
471        Therefore, you must view this method for each mix-ins to determine which
472        kwargs must be passed.
473
474        #### Parameters
475
476        - `as_dict` :
477            Whether to return the populated data as a dictionary or to initialize
478            it as a database object. Defaults to False.
479        - `**kwargs` :
480            All fields required in order to initialize the database entry. For
481            example, a Structure table would require a toolkit structure, while
482            a Thermodynamics table would require an energy.
483
484        Returns
485        -------
486        a dictionary if as_dict is True; and a database object if as_dict is False
487        """
488
489        # Grab a list of all parent classes
490        parents = inspect.getmro(cls)
491
492        # As we go through the parent classes below, we will identify the mixins
493        # and populate data using those classes. All of this is fed in to our
494        # main class at the end of the function. We keep this running dictionary
495        # as we go.
496        all_data = kwargs.copy()
497        # TODO: How should I best handle passing extra kwargs to the final class
498        # initialization? For example, I would want to pass `energy` but I wouldn't
499        # want to pass the toolkit structure object. I may update this line in
500        # the future to remove python objects. For now, I only remove structure
501        # and migration_hop because I know that it is a toolkit object -- not
502        # a database column.
503        all_data.pop("structure", None)
504        all_data.pop("migration_hop", None)
505        all_data.pop("band_structure", None)
506        all_data.pop("density_of_states", None)
507
508        for parent in parents:
509
510            # Skip the parent class if it doesn't directly inherit from the
511            # DatabaseTable class. We do this because we want the fundamental
512            # mixins that come with Simmate (such as Structure, Forces, Thermodynamics).
513            # We also skip the classes that don't have a _from_toolkit method defined.
514            # Right now, this is just the Calculation class that's skipped with this
515            # condition.
516            if parent.__base__ != DatabaseTable or not hasattr(parent, "_from_toolkit"):
517                continue
518
519            # Grab the input arguments for the _from_toolkit method. This will
520            # give a list back like... ['cls', 'structure', 'as_dict']. We
521            # don't consider as_dict or cls, so we remove those too
522            inputs = inspect.getfullargspec(parent._from_toolkit).args
523            inputs.remove("as_dict")
524            inputs.remove("cls")
525
526            # Now inputs is current of list of keys that we need. Next, we go
527            # through our kwargs and see if we have any of these keys, and if so,
528            # grab them.
529            matching_inputs = {
530                key: kwargs[key] for key in inputs if key in kwargs.keys()
531            }
532
533            # We now pass these inputs to the _from_toolkit. If we're missing
534            # a required input, this will raise an error here. We only
535            # want the expanded input, so we request a dictionary, not object.
536            data = parent._from_toolkit(**matching_inputs, as_dict=True)
537
538            # Now add this mixin's data to our collective dictionary
539            all_data.update(data)
540
541        # If as_dict is false, we build this into an Object. Otherwise, just
542        # return the dictionary
543        return all_data if as_dict else cls(**all_data)

545    @classmethod
546    def _confirm_override(
547        cls,
548        confirm_override: bool,
549        parallel: bool,
550        confirm_sqlite_parallel: bool,
551    ):
552        """
553        A utility to make sure the user wants to load new data into their table
554        and (if they are using sqlite) that they are aware of the risks of
555        parallelizing their loading.
556
557        This utility should not be called directly, as it is used within
558        load_archive and load_remote_archive.
559        """
560        # first check if the table has data in it already. We raise errors
561        # to stop the user from doing unneccessary and potentiall destructive
562        # downloads
563        if cls.objects.exists() and not confirm_override:
564            # if the user has a third-party app, we can be more specific with
565            # our error message.
566            if cls._meta.app_label == "third_parties":
567                raise Exception(
568                    "It looks like you're using a third-party database table and "
569                    "that the table already has data in it! This means you already "
570                    "called load_remote_archive and don't need to do it again. "
571                    "If you are trying reload a newer version of this data, make "
572                    "sure you empty this table first. This can be done by "
573                    "reseting your database or manually deleting all objects "
574                    "with `ExampleTable.objects.all().delete()`"
575                )
576
577            # otherwise warning the user of overwriting data with matching
578            # primary keys -- and ask them to use confirm_override.
579            raise Exception(
580                "It looks like this table already has data in it! By loading an "
581                "archive, you could potentially overwrite this data. The most "
582                "common mistake is non-unique primary keys between your current "
583                "data and the archive -- if there is a duplicate primary key, it "
584                "will overwrite your data. If you are confident the data is safe "
585                "to load into your database, run this command again with "
586                "confirm_override=True."
587            )
588
589        # Django and Dask can only handle so much for the parallelization
590        # of database writing with SQLite. So if the user has SQLite as their
591        # backend, we need to stop them from using this feature.
592        from simmate.configuration.django.settings import DATABASES
593
594        if parallel and not confirm_sqlite_parallel and "sqlite3" in str(DATABASES):
595            raise Exception(
596                "It looks like you are trying to run things in parallel but are "
597                "using the default database backend (sqlite3), which is not "
598                "always stable for massively parallel methods. You can still "
599                "do this, but this message serves as a word of caution. "
600                "You If you see error messages pop up saying 'database is "
601                "locked', then your database is not stable at the rate you're "
602                "trying to write data. This is a sign that you should either "
603                "(1) switch to a different database backend such as Postgres "
604                "or (2) reduce the parallelization of your tasks. If you are "
605                "comfortable with these warnings and know what you're doing, "
606                "set confirm_sqlite_parallel=True."
607            )
608
609    # @transaction.atomic  # We can't have an atomic transaction if we use Dask
610    @classmethod
611    def load_archive(
612        cls,
613        filename: str = None,
614        delete_on_completion: bool = False,
615        confirm_override: bool = False,
616        parallel: bool = False,
617        confirm_sqlite_parallel: bool = False,
618    ):
619        """
620        Reads a compressed zip file made by `objects.to_archive` and loads the data
621        back into the Simmate database.
622
623        Typically, users won't call this method directly, but instead use the
624        `load_remote_archive` method, which handles downloading the archive
625        file from the Simmate website for you.
626
627        #### Parameters
628
629        - `filename`:
630            The filename to write the zip file to. By defualt, None will try to
631            find a file named "MyExampleTableName-2022-01-25.zip", where the date
632            corresponds to version/timestamp. If multiple files match this format
633            the most recent date will be used.
634
635        - `delete_on_completion`:
636            Whether to delete the archive file once all data is loaded into the
637            database. Defaults to False
638
639        - `confirm_override`:
640            If the table already has data in it, the user must take particular
641            care to downloading new data. This flag makes sure the user has
642            made the proper checks to run this action. Default is False.
643
644        - `parallel`:
645            Whether to load the data in parallel. If true, this will start
646            a local Dask cluster and each data row will be submitted as a task
647            to the cluster. This provides substansial speed-ups for loading
648            large datasets into the dataset. Default is False.
649
650        - `confirm_sqlite_parallel`:
651            If the database backend is sqlite, this parameter ensures the user
652            knows what they are doing and know the risks of parallelization.
653            Default is False.
654        """
655
656        # make sure the user actually wants to do this!
657        cls._confirm_override(
658            confirm_override,
659            parallel,
660            confirm_sqlite_parallel,
661        )
662
663        from tqdm import tqdm
664        from simmate.toolkit import Structure as ToolkitStructure
665
666        # generate the file name if one wasn't given
667        if not filename:
668            # The name will be something like "MyExampleTable-2022-01-25.zip".
669            # We go through all files that match "MyExampleTable-*.zip" and then
670            # grab the most recent date.
671            matching_files = [
672                file
673                for file in os.listdir()
674                if file.startswith(cls.__name__) and file.endswith(".zip")
675            ]
676            # make sure there is at least one file
677            if not matching_files:
678                raise FileNotFoundError(
679                    f"No file found matching the {cls.__name__}-*.zip format"
680                )
681            # sort the files by date and grab the first
682            matching_files.sort(reverse=True)
683            filename = matching_files[0]
684
685        # Turn the filename into the full path -- which makes a number of
686        # manipulations easier below.
687        filename = os.path.abspath(filename)
688
689        # uncompress the zip file to the same directory that it is located in
690        shutil.unpack_archive(
691            filename,
692            extract_dir=os.path.dirname(filename),
693        )
694
695        # We will now have a csv file of the same name, which we load into
696        # a pandas dataframe
697        csv_filename = filename.replace(".zip", ".csv")
698        df = pandas.read_csv(csv_filename)
699
700        # BUG: NaN values throw errors when read into SQL databases, so we
701        # convert all NaN entries to None. This hacky line was taken from
702        #   https://stackoverflow.com/questions/39279824/
703        df = df.astype(object).where(df.notna(), None)
704
705        # convert the dataframe to a list of dictionaries that we will iterate
706        # through.
707        entries = df.to_dict(orient="records")
708
709        # to enable parallelization, we define a function to load a single
710        # entry (or row) of data. This allows us to submit the function to Dask.
711        def load_single_entry(entry):
712
713            # For all entries, convert the structure_string to a toolkit structure
714            if "structure_string" in entry:
715                structure_str = entry.pop("structure_string")
716                structure = ToolkitStructure.from_database_string(structure_str)
717                entry["structure"] = structure
718            # OPTIMIZE: is there a better way to do decide which entries need to be
719            # converted to toolkit objects? This code may be better placed in
720            # base_data_type methods (e.g. a `from_base_info` method for each)
721
722            entry_db = cls.from_toolkit(**entry)
723            entry_db.save()
724
725        # now iterate through all entries to save them to the database
726        if not parallel:
727            # If user doesn't want parallelization, we run these in the main
728            # thread and monitor progress with tqdm
729            for entry in tqdm(entries):
730                load_single_entry(entry)
731        # otherwise we use dask to submit these in batches!
732        else:
733
734            from simmate.configuration.dask import batch_submit
735
736            batch_submit(
737                function=load_single_entry,
738                args_list=entries,
739                batch_size=15000,
740            )
741
742        # We can now delete the files. The zip file is only deleted if requested.
743        os.remove(csv_filename)
744        if delete_on_completion:
745            os.remove(filename)  # the zip archive
746
747    @classmethod
748    def load_remote_archive(
749        cls,
750        remote_archive_link: str = None,
751        confirm_override: bool = False,
752        parallel: bool = False,
753        confirm_sqlite_parallel: bool = False,
754    ):
755        """
756        Downloads a compressed zip file made by `objects.to_archive` and loads
757        the data back into the Simmate database.
758
759        This method should only be called once -- when you have a completely
760        empty database. After this call, all data will be stored locally and
761        you don't need to call this method again (even accross python sessions).
762
763        #### Parameters
764
765        - `remote_archive_link`:
766            The URL for that the archive will be downloaded from. If not supplied,
767            it will default to the table's remote_archive_link attribute.
768
769        - `confirm_override`:
770            If the table already has data in it, the user must take particular
771            care to downloading new data. This flag makes sure the user has
772            made the proper checks to run this action.
773
774        - `parallel`:
775            Whether to load the data in parallel. If true, this will start
776            a local Dask cluster and each data row will be submitted as a task
777            to the cluster. This provides substansial speed-ups for loading
778            large datasets into the dataset. Default is False.
779
780        - `confirm_sqlite_parallel`:
781            If the database backend is sqlite, this parameter ensures the user
782            knows what they are doing and know the risks of parallelization.
783            Default is False.
784        """
785
786        # make sure the user actually wants to do this!
787        cls._confirm_override(
788            confirm_override,
789            parallel,
790            confirm_sqlite_parallel,
791        )
792
793        # confirm that we have a link to download from
794        if not remote_archive_link:
795            # if no link was given we take the input value from class attribute
796            if not cls.remote_archive_link:
797                raise Exception(
798                    "This table does not have a default link to load the archive "
799                    " from. You must provide a remote_archive_link."
800                )
801            remote_archive_link = cls.remote_archive_link
802
803        # tell the user where the data comes from
804        if cls._meta.app_label == "third_parties":
805            print(
806                "\nWARNING: this data is NOT from the Simmate team, so be sure "
807                "to visit the provider's website and to cite their work."
808                f" This data is from {cls.source} and the following paper "
809                f"should be cited: {cls.source_doi}\n"
810            )
811
812        # Predetermine the file name, which is just the ending of the URL
813        archive_filename = remote_archive_link.split("/")[-1]
814
815        # Download the archive zip file from the URL to the current working dir
816        print("Downloading archive file...")
817        urllib.request.urlretrieve(remote_archive_link, archive_filename)
818        print("Done.\n")
819
820        # now that the archive is downloaded, we can load it into our db
821        print("Loading data into Simmate database...")
822        cls.load_archive(
823            archive_filename,
824            delete_on_completion=True,
825            confirm_override=True,  # we already confirmed this above
826            parallel=parallel,
827            confirm_sqlite_parallel=True,  # we already confirmed this above
828        )
829        print("Done.\n")

887        """
888        Finds all columns that aren't covered by the supported Table mix-ins.
889
890        For example, a table made from...
891
892        ``` python
893        from simmate.database.base_data_types import (
894            table_column,
895            Structure,

898
899        class ExampleTable(Structure, Forces):

901            custom_column2 = table_column.FloatField()

903
904        ... would return ...
905

907        ["custom_column1", "custom_column2"]
908        ```
909        """

912        columns_w_mixin = [
913            column for mixin in cls.get_mixins() for column in mixin.get_column_names()
914        ]
915        extra_columns = [
916            column
917            for column in all_columns
918            if column not in columns_w_mixin and column != "id"
919        ]
