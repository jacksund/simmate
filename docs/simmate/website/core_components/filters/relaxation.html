162    def create_subclasses(cls, name: str, module: str, **extra_columns):
163        """
164        Dynamically creates a subclass of Relaxation as well as a separate IonicStep
165        table for it. These tables are linked together.
166
167        Example use:
168
169        ``` python
170        from simmate.database.base_data_types import Relaxation
171
172        ExampleRelaxation, ExampleIonicStep = Relaxation.create_subclasses(
173            "Example",
174            module=__name__,
175        )
176        ```
177
178        #### Parameters
179
180        - `name` :
181            The prefix name of the subclasses that are output. "Relaxation" and
182            "IonicStep" will be attached to the end of this prefix.
183        - `module` :
184            name of the module this subclass should be associated with. Typically,
185            you should pass __name__ to this.
186        - `**extra_columns` :
187            Additional columns to add to the table. The keyword will be the
188            column name and the value should match django options
189            (e.g. table_column.FloatField())
190
191        #### Returns
192
193        - `NewRelaxationClass` :
194            A subclass of Relaxation.
195        - `NewIonicStepClass`:
196            A subclass of IonicStep.
197
198        """
199
200        # For convience, we add columns that point to the start and end structures
201        NewRelaxationClass = cls.create_subclass(
202            f"{name}Relaxation",
203            structure_start=table_column.OneToOneField(
204                f"{name}IonicStep",
205                on_delete=table_column.CASCADE,
206                related_name="relaxations_as_start",
207                blank=True,
208                null=True,
209            ),
210            structure_final=table_column.OneToOneField(
211                f"{name}IonicStep",
212                on_delete=table_column.CASCADE,
213                related_name="relaxations_as_final",
214                blank=True,
215                null=True,
216            ),
217            module=module,
218            **extra_columns,
219        )
220
221        NewIonicStepClass = IonicStep.create_subclass_from_relaxation(
222            name,
223            NewRelaxationClass,
224            module=module,
225            **extra_columns,
226        )
227
228        # we now have a new child class and avoided writing some boilerplate code!
229        return NewRelaxationClass, NewIonicStepClass

263    def update_from_vasp_run(
264        self,
265        vasprun: Vasprun,
266        corrections: List,
267        directory: str,
268    ):
269        """
270        Given a Vasprun object from a finished relaxation, this will update the
271        Relaxation table entry and the corresponding IonicStep entries.
272
273        #### Parameters
274
275        vasprun :
276            The final Vasprun object from the relaxation outputs.
277        corrections :
278            List of errors and corrections applied to during the relaxation.
279        directory :
280            name of the directory that relaxation was ran in. This is only used
281            to reference the archive file if it's ever needed again.
282        """
283
284        # The data is actually easier to access as a dictionary and everything
285        # we need is stored under the "output" key
286        data = vasprun.as_dict()["output"]
287
288        # The only other data we need to grab is the list of structures. We can
289        # pull the structure for each ionic step from the vasprun class directly.
290        structures = vasprun.structures
291
292        # Now let's iterate through the ionic steps and save these to the database.
293        for number, (structure, ionic_step) in enumerate(
294            zip(structures, data["ionic_steps"])
295        ):
296            # first pull all the data together and save it to the database. We
297            # are saving this to an IonicStepStructure datatable. To access this
298            # model, we look need to use "structures.model".
299            structure = self.structures.model.from_toolkit(
300                number=number,
301                structure=structure,
302                energy=ionic_step["e_wo_entrp"],
303                site_forces=ionic_step["forces"],
304                lattice_stress=ionic_step["stress"],
305                relaxation=self,  # this links the structure to this relaxation
306            )
307            structure.save()
308
309            # If this is the first structure, we want to link it as such
310            if number == 0:
311                self.structure_start_id = structure.id
312            # and same for the final structure. Note, we can't use elif becuase
313            # there's a chance the start/end structure are the same, which occurs
314            # when the starting structure is found to be relaxed already.
315            if number == len(structures) - 1:
316                self.structure_final_id = structure.id
317
318        # update our relaxation entry with new data
319        self.update_from_toolkit(
320            # use the final ionic setup for the structure and energy
321            structure=structures[-1],
322            energy=data["ionic_steps"][-1]["e_wo_entrp"],
323            # calculate extra data for storing
324            volume_change=structures[-1].volume
325            - structures[0].volume / structures[0].volume,
326            # There is also extra data for the final structure that we save directly
327            # in the relaxation table.  We use .get() in case the key isn't provided.
328            band_gap=data.get("bandgap"),
329            is_gap_direct=data.get("is_gap_direct"),
330            energy_fermi=data.get("efermi"),
331            conduction_band_minimum=data.get("cbm"),
332            valence_band_maximum=data.get("vbm"),
333            # lastly, we also want to save the corrections made and directory it ran in
334            corrections=corrections,
335            directory=directory,
336        )

409class IonicStep(Structure, Thermodynamics, Forces):
410    """
411    This database table that holds the data for each ionic step of a relaxation.
412
413    An ionic step can be viewed as a static energy calculation, but we keep these
414    results separate because:
415
416        1. Pulay stress can make these energies/forces inaccurate
417        2. These results each have an associated Relaxation and ionic step number
418
419    You will likely never access this table directly. Instead, data is better
420    accessed through the `structures` attribute on appropiate Relaxation table.
421
422    For example:
423
424    ``` python
425    from simmate.database import connect
426    from simmate.database.workflow_results import MITRelaxation
427
428    # grab your desired relaxation
429    relax = MITRelaxation.objects.get(id=1)
430
431    # grab the associated ionic steps
432    ionic_steps = relax.structures.all()
433    ```
434
435    Note: we assume that only converged data is being stored! So there is no
436    "converged_electronic" column here. ErrorHandlers and workups should
437    ensure convergence.
438    """
439

This database table that holds the data for each ionic step of a relaxation.

An ionic step can be viewed as a static energy calculation, but we keep these results separate because:

1. Pulay stress can make these energies/forces inaccurate
2. These results each have an associated Relaxation and ionic step number

You will likely never access this table directly. Instead, data is better accessed through the structures attribute on appropiate Relaxation table.

For example:

from simmate.database import connect
from simmate.database.workflow_results import MITRelaxation

# grab your desired relaxation
relax = MITRelaxation.objects.get(id=1)

# grab the associated ionic steps
ionic_steps = relax.structures.all()

Note: we assume that only converged data is being stored! So there is no "converged_electronic" column here. ErrorHandlers and workups should ensure convergence.
