 20class DynamicsRun(Structure, Calculation):
 21    """
 22    Holds results from a dynamics simulations -- often referred to as a molecular
 23    dynamics run.
 24
 25    In addition to the attributes listed, you can also access all ionic steps
 26    of the run via the `structures` attribute. This attribute gives a list of
 27    `DynamicsIonicSteps`.
 28    """
 29
 30    class Meta:
 31        abstract = True
 32        app_label = "workflows"
 33
 34    temperature_start = table_column.IntegerField(blank=True, null=True)
 35    """
 36    The starting tempertature of the simulation in Kelvin.
 37    """
 38
 39    temperature_end = table_column.IntegerField(blank=True, null=True)
 40    """
 41    The ending tempertature of the simulation in Kelvin.
 42    """
 43
 44    time_step = table_column.FloatField(blank=True, null=True)
 45    """
 46    The time in picoseconds for each step in the simulation.
 47    """
 48
 49    nsteps = table_column.IntegerField(blank=True, null=True)
 50    """
 51    The total number of steps in the simulation. Note, this is the maximum cutoff
 52    steps set. In some cases, there may be fewer steps when the simulation is 
 53    stopped early.
 54    """
 55
 56    @classmethod
 57    def create_subclasses(cls, name: str, module: str, **extra_columns):
 58        """
 59        Dynamically creates a subclass of DynamicsRun as well as a separate
 60        DynamicsIonicStep table for it. These tables are linked together.
 61
 62        Example use:
 63
 64        ``` python
 65        from simmate.database.base_data_types import DynamicsRun
 66
 67        ExampleDynamicsRun, ExampleDynamicsIonicStep = DynamicsRun.create_subclasses(
 68            "Example",
 69            module=__name__,
 70        )
 71        ```
 72
 73        #### Parameters
 74
 75        - `name` :
 76            The prefix name of the subclasses that are output. "DynamicsRun" and
 77            "DynamicsIonicStep" will be attached to the end of this prefix.
 78        - `module` :
 79            name of the module this subclass should be associated with. Typically,
 80            you should pass __name__ to this.
 81        - `**extra_columns` :
 82            Additional columns to add to the table. The keyword will be the
 83            column name and the value should match django options
 84            (e.g. table_column.FloatField())
 85
 86        Returns
 87        -------
 88        - `NewDynamicsRunClass` :
 89            A subclass of DynamicsRun.
 90        - `NewDynamicsIonicStepClass`:
 91            A subclass of DynamicDynamicsIonicStep.
 92
 93        """
 94
 95        # For convience, we add columns that point to the start and end structures
 96        NewDynamicsRunClass = cls.create_subclass(
 97            f"{name}DynamicsRun",
 98            module=module,
 99            **extra_columns,
100        )
101
102        NewDynamicDynamicsIonicStepClass = (
103            DynamicsIonicStep.create_subclass_from_dynamics_run(
104                name,
105                NewDynamicsRunClass,
106                module=module,
107                **extra_columns,
108            )
109        )
110
111        # we now have a new child class and avoided writing some boilerplate code!
112        return NewDynamicsRunClass, NewDynamicDynamicsIonicStepClass

114    def update_from_vasp_run(
115        self,
116        vasprun: Vasprun,
117        corrections: List,
118        directory: str,
119    ):
120        """
121        Given a Vasprun object from a finished dynamics run, this will update the
122        DynamicsRun table entry and the corresponding DynamicsIonicStep entries.
123
124        #### Parameters
125
126        vasprun :
127            The final Vasprun object from the dynamics run outputs.
128        corrections :
129            List of errors and corrections applied to during the relaxation.
130        directory :
131            name of the directory that relaxation was ran in. This is only used
132            to reference the archive file if it's ever needed again.
133        """
134
135        # The data is actually easier to access as a dictionary and everything
136        # we need is stored under the "output" key
137        data = vasprun.as_dict()["output"]
138
139        # The only other data we need to grab is the list of structures. We can
140        # pull the structure for each ionic step from the vasprun class directly.
141        structures = vasprun.structures
142
143        # Now let's iterate through the ionic steps and save these to the database.
144        for number, (structure, ionic_step) in enumerate(
145            zip(structures, data["ionic_steps"])
146        ):
147            # first pull all the data together and save it to the database. We
148            # are saving this to an DynamicsIonicStepStructure datatable. To access this
149            # model, we look need to use "structures.model".
150            structure = self.structures.model.from_toolkit(
151                number=number,
152                structure=structure,
153                energy=ionic_step["e_wo_entrp"],
154                site_forces=ionic_step["forces"],
155                lattice_stress=ionic_step["stress"],
156                temperature=self._get_temperature_at_step(number),
157                dynamics_run=self,  # this links the structure to this dynamics run
158            )
159            structure.save()
160
161        # lastly, we also want to save the corrections made and directory it ran in
162        self.corrections = corrections
163        self.directory = directory
164
165        # Now we have the relaxation data all loaded and can save it to the database
166        self.save()
167
168    def _get_temperature_at_step(self, step_number: int):
169        return step_number * self._get_temperature_step_size() + self.temperature_start
170
171    def _get_temperature_step_size(self):
172        return (self.temperature_end - self.temperature_start) / self.nsteps

In addition to the attributes listed, you can also access all ionic steps of the run via the structures attribute. This attribute gives a list of DynamicsIonicSteps.

175class DynamicsIonicStep(Structure, Thermodynamics, Forces):
176    """
177    Holds information for a single ionic step of a `DynamicsRun`.
178
179    Each entry will map to a `DynamicsRun`, so you should typically access this
180    data through that class. The exception to this is when you want all ionic
181    steps accross many relaxations for a machine learning input.
182    """
183
184    class Meta:
185        abstract = True
186        app_label = "workflows"
187
188    base_info = (
189        ["number"] + Structure.base_info + Thermodynamics.base_info + Forces.base_info

192    number = table_column.IntegerField()

194    This is ionic step number for the given relaxation. This starts counting from 0.
195    """

198    """
199    Expected temperature based on the temperature_start/end and nsteps. Note
200    that in-practice some steps may not be at equilibrium temperatue. This could
201    be the 1st 100 steps of a run or alternatively when the thermostat component
202    is off-temperature.
203    """

213    @classmethod
214    def create_subclass_from_dynamics_run(
215        cls,
216        name: str,
217        dynamics_run: DynamicsRun,
218        module: str,
219        **extra_columns,
220    ):
221        """
222        Dynamically creates a subclass of DynamicsIonicStep and links it to the
223        DynamicsRun table.
224
225        This method should NOT be called directly because it is instead used by
226        `DynamicsRun.create_subclasses`.
227
228        #### Parameters
229
230        - `name` :
231            Name of the subclass that is output.
232        - `dynamics_run` :
233            DynamicsRun table that these ionic steps should be associated with.
234        - `module` :
235            name of the module this subclass should be associated with. Typically,
236            you should pass __name__ to this.
237        - `**extra_columns` :
238            Additional columns to add to the table. The keyword will be the
239            column name and the value should match django options
240            (e.g. table_column.FloatField())
241
242        Returns
243        -------
244        - `NewClass` :
245            A subclass of DynamicsIonicStep.
246
247        """
248
249        # All structures in this table come from dynamics run calculations, where
250        # there can be many structures (one for each ionic step) linked to a
251        # single run. This means the start structure, end structure, and
252        # those structure in-between are stored together here.
253        # Therefore, there's just a simple column stating which relaxation it
254        # belongs to.
255        NewClass = cls.create_subclass(
256            f"{name}DynamicsIonicStep",
257            dynamics_run=table_column.ForeignKey(
258                dynamics_run,
259                on_delete=table_column.CASCADE,
260                related_name="structures",
261            ),
262            module=module,
263            **extra_columns,
264        )
265
266        # we now have a new child class and avoided writing some boilerplate code!
267        return NewClass
