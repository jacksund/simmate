  9    """
 10    A calculation is the result of a single `workflow.run()` call. Thus, a
 11    calculation is synonmyous with a Prefect "Flow-Run". Becuase of this,
 12    every table that inherits from this Calculation class will be directly
 13    linked to a specific Workflow. You can access this workflow via the
 14    `workflow` attribute.
 15    """

 19        app_label = "workflows"

 29    directory = table_column.CharField(
 30        max_length=250,
 31        blank=True,
 32        null=True,
 33    )
 34    """
 35    This is the folder that the workflow was ran in. It will be something
 36    like... "/path/to/simmate-task-abc123"
 37    """
 38    # TODO: maybe add host computer name too? (via platform.node())
 39
 40    prefect_flow_run_id = table_column.CharField(
 41        max_length=50,
 42        blank=True,
 43        null=True,
 44        unique=True,
 45    )
 46    """
 47    If this was submitted through Prefect, there is a lot more metadata available
 48    for this calculation. Simmate does not store this data directly, but instead
 49    we store the flow-run-id so that the user may look this up in the Prefect
 50    database if they wish.
 51    An example id is... d8a785e1-e344-463a-bede-0e7b3da7bab6
 52    """

 64    corrections = table_column.JSONField(blank=True, null=True)
 65    """
 66    Simmate workflows often have ErrorHandlers that fix any issues while the

 68    any of those changes here.
 69    """

 71    @property
 72    def prefect_cloud_link(self) -> str:
 73        """
 74        URL to this calculation (flow-run) in the Prefect Cloud website.
 75
 76        This assumes that the calculation was registered with prefect cloud and
 77        doesn't check to confirm it's been registered. To actually confirm that,
 78        use the `flow_run_view` attribute instead.
 79        """
 80        return f"https://cloud.prefect.io/simmate/flow-run/{self.prefect_flow_run_id}"
 81
 82    @property
 83    def flow_run_view(self) -> FlowRunView:
 84        """
 85        Checks if the prefect_flow_run_id was registered with Prefect Cloud, and
 86        if so, returns a
 87        [FlowRunView](https://docs.prefect.io/orchestration/flow-runs/inspection.html)
 88        that hold metadata such as the status. This metadata includes...
 89            - agent_id
 90            - auto_scheduled
 91            - context
 92            - created_by_user_id
 93            - end_time
 94            - flow_id
 95            - labels
 96            - name
 97            - parameters
 98            - scheduled_start_time
 99            - start_time
100            - state
101            - tenant_id
102            - times_resurrected
103            - version
104
105        If Prefect Cloud is not configured or if the calculation was ran
106        locally, the None is returned.
107        """
108        try:
109            return FlowRunView.from_flow_run_id(self.prefect_flow_run_id)
110        except:  # may fail if this is a local run or prefect api key not configured
111            return None
112
113    @property
114    def prefect_flow_run_name(self) -> str:
115        """
116        Gives the user-friendly name of this run if the prefect_flow_run_id
117        was registered with Prefect Cloud. (an example name is "friendly-bumblebee").
118        """
119        flowrunview = self.flow_run_view
120        return flowrunview.name if flowrunview else None
121
122    @property
123    def prefect_state(self) -> str:
124        """
125        Gives the current state of this run if the prefect_flow_run_id
126        was registered with Prefect Cloud. (ex: "Scheduled" or "Successful")
127        """
128        flowrunview = self.flow_run_view
129        return flowrunview.state.__class__.__name__ if flowrunview else None
130
131    @classmethod
132    def from_prefect_id(cls, id: str, **kwargs):
133        """
134        Given a prefect id, this method will do one of the following...
135
136        1. if it exists, load the database object with matching prefect_id
137        2. if it doesn't exist, create a new database object using this ID and extra kwargs
138
139        It will then return the corresponding Calculation instance.
140        """
141
142        # Depending on how a workflow was submitted, there may be a calculation
143        # extry existing already -- which we need to grab and then update. If it's
144        # not there, we create a new one!
145
146        # check if the calculation already exists in our database, and if so,
147        # grab it and return it.
148        if cls.objects.filter(prefect_flow_run_id=id).exists():
149            return cls.objects.get(prefect_flow_run_id=id)
150        # Otherwise we need to create a new one and return that.
151
152        # To handle the initialization of other Simmate mix-ins, we pass all
153        # information to the from_toolkit method rather than directly to cls.
154        calculation = cls.from_toolkit(prefect_flow_run_id=id, **kwargs)
155        calculation.save()
156
157        return calculation
