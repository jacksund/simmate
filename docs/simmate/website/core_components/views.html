 23def structure_viewer(request):
 24
 25    # Grabs all data after the '?' in the URL
 26    query = request.GET.dict()
 27
 28    # convert the query to a Structure object
 29    structure_string = query.get("structure_string", "")
 30    structure = Structure.from_database_string(structure_string)
 31
 32    # We want to make a 3d structure file. We make this so it...
 33    #   1. has a random name that ends with ".glb"
 34    #   2. is a temporary file (deletes once finished)
 35    #   3. is located in the static root folder
 36    # Note we make this file in the static directory because we want to use
 37    # "load_static" to grab it in the template
 38    temp_dir = get_directory(
 39        settings.STATIC_ROOT
 40    )  # use get_dir to ensure folder exists
 41    temp_file = tempfile.NamedTemporaryFile(
 42        dir=temp_dir,
 43        suffix=".glb",
 44        # BUG (the fix is below): if delete=True, this sets up a race condition
 45        # of when the user recieves the file versus when it is deleted.
 46        delete=False,
 47    )
 48    temp_filname_base = os.path.basename(temp_file.name)
 49
 50    # BUG FIX: Because we can make these temporary files automatically delete,
 51    # we need to prevent 3D files from building up over time and taking up
 52    # room on the server/disk. We therefore automatically delete any files that
 53    # are older than N seconds. This process can be slow, but it adds minimal
 54    # overhead to the blender-creation process.
 55    detete_old_3d_files()
 56
 57    # Use blender to create a temporary glb file. Note, the output here
 58    # may be useful for debugging, but it isn't used at the moment.
 59    output = make_blender_structure(structure, filename=temp_file.name)
 60
 61    # we pass the base name to the template so that it knows where the static
 62    # file is located (template assumes static directory)
 63    context = {"3d_structure_filename": temp_filname_base}
 64    template = "core_components/structure_viewer.html"
 65    return render(request, template, context)

 87def detete_old_3d_files(time_cutoff: float = 60):
 88    """
 89    Goes through the static directory and finds all "tmp***.glb" files that
 90    are older than a given time cutoff. Each of these files is then deleted.
 91
 92    #### Parameters
 93
 94    - `time_cutoff`:
 95        The time (in seconds) required to determine whether a file is old or not.
 96        The default is 60 seconds.
 97    """
 98    # load the full path to the desired directory
 99    directory = os.path.join(settings.DJANGO_DIRECTORY, "static")
100
101    # grab all files/folders in this directory and then limit this list to those
102    # that are...
103    #   1. NOT folders
104    #   2. start with "tmp"
105    #   3. end with ".glb"
106    #   3. haven't been modified for at least time_cutoff
107    filenames = []
108    for filename in os.listdir(directory):
109        filename_full = os.path.join(directory, filename)
110        if (
111            not os.path.isdir(filename_full)
112            and filename.startswith("tmp")
113            and filename.endswith(".glb")
114            and time.time() - os.path.getmtime(filename_full) > time_cutoff
115        ):
116            filenames.append(filename_full)
117
118    # now go through this list and delete the files that met the criteria.
119    # If the file fails to be deleted, we just ingore it and move on
120    for filename in filenames:
121        try:
122            os.remove(filename)
123        except:
124            continue
