simmate.file_converters.structure.ase

This converter simply imports a pymatgen class. With this you can quickly convert between Simmate.toolkit and ASE.atoms objects.

from simmate.toolkit import Structure
from simmate.file_converters.structure.ase import AseAtomsAdaptor


initial_structure = Structure.from_file("example.cif")

# convert to ASE
new_ase_object = AseAtomsAdaptor.get_atoms(structure)

# convert back to PyMatgen/Simmate
new_toolkit_object = AseAtomsAdaptor.get_structure(new_ase_object)

# alternatively, you can convert back to PyMatgen Molecule
new_molecule_object = AseAtomsAdaptor.get_molecule(new_ase_object)

 40    @staticmethod
 41    def get_atoms(structure, **kwargs):
 42        """
 43        Returns ASE Atoms object from pymatgen structure or molecule.
 44
 45        Args:
 46            structure: pymatgen.core.structure.Structure or pymatgen.core.structure.Molecule
 47            **kwargs: other keyword args to pass into the ASE Atoms constructor
 48
 49        Returns:
 50            ASE Atoms object
 51        """
 52        if not structure.is_ordered:
 53            raise ValueError("ASE Atoms only supports ordered structures")
 54        if not ase_loaded:
 55            raise ImportError(
 56                "AseAtomsAdaptor requires the ASE package.\nUse `pip install ase` or `conda install ase -c conda-forge`"
 57            )
 58
 59        # Construct the base ASE Atoms object
 60        symbols = [str(site.specie.symbol) for site in structure]
 61        positions = [site.coords for site in structure]
 62        if hasattr(structure, "lattice"):
 63            cell = structure.lattice.matrix
 64            pbc = True
 65        else:
 66            cell = None
 67            pbc = None
 68
 69        atoms = Atoms(symbols=symbols, positions=positions, pbc=pbc, cell=cell, **kwargs)
 70
 71        # Set the site magmoms in the ASE Atoms object
 72        # Note: ASE distinguishes between initial and converged
 73        # magnetic moment site properties, whereas pymatgen does not. Therefore, we
 74        # have to distinguish between these two when constructing the Structure/Molecule.
 75        # The mapping selected here is:
 76        # ASE initial magmom <--> Pymatgen "magmom"
 77        # ASE final magmom <--> Pymatgen "final_magmom"
 78        # ASE initial charge <--> Pymatgen "charge"
 79        # ASE final charge <--> Pymatgen "final_charge"
 80        if "magmom" in structure.site_properties:
 81            initial_magmoms = structure.site_properties["magmom"]
 82            atoms.set_initial_magnetic_moments(initial_magmoms)
 83        if "charge" in structure.site_properties:
 84            initial_charges = structure.site_properties["charge"]
 85            atoms.set_initial_charges(initial_charges)
 86
 87        if "final_magmom" in structure.site_properties:
 88            magmoms = structure.site_properties["final_magmom"]
 89        else:
 90            magmoms = None
 91        if "final_charge" in structure.site_properties:
 92            charges = structure.site_properties["final_charge"]
 93        else:
 94            charges = None
 95        if magmoms or charges:
 96            if magmoms and charges:
 97                calc = SinglePointDFTCalculator(atoms, **{"magmoms": magmoms, "charges": charges})
 98            elif magmoms:
 99                calc = SinglePointDFTCalculator(atoms, **{"magmoms": magmoms})
100            elif charges:
101                calc = SinglePointDFTCalculator(atoms, **{"charges": charges})
102            atoms.calc = calc
103
104        # Get the oxidation states from the structure
105        oxi_states = [getattr(site.specie, "oxi_state", None) for site in structure]
106
107        # Read in selective dynamics if present. Note that the ASE FixAtoms class fixes (x,y,z), so
108        # here we make sure that [False, False, False] or [True, True, True] is set for the site selective
109        # dynamics property. As a consequence, if only a subset of dimensions are fixed, this won't get passed to ASE.
110        if "selective_dynamics" in structure.site_properties:
111            fix_atoms = []
112            for site in structure:
113                site_prop = site.properties["selective_dynamics"]
114                if site_prop not in [[True, True, True], [False, False, False]]:
115                    raise ValueError(
116                        "ASE FixAtoms constraint does not support selective dynamics in only some dimensions."
117                        "Remove the selective dynamics and try again if you do not need them."
118                    )
119                is_fixed = bool(~np.all(site.properties["selective_dynamics"]))
120                fix_atoms.append(is_fixed)
121
122        else:
123            fix_atoms = None
124
125        # Set the selective dynamics with the FixAtoms class.
126        if fix_atoms is not None:
127            atoms.set_constraint(FixAtoms(mask=fix_atoms))
128
129        # Add any remaining site properties to the ASE Atoms object
130        for prop in structure.site_properties:
131            if prop not in ["magmom", "charge", "final_magmom", "final_charge", "selective_dynamics"]:
132                atoms.set_array(prop, np.array(structure.site_properties[prop]))
133        if np.any(oxi_states):
134            atoms.set_array("oxi_states", np.array(oxi_states))
135
136        return atoms

138    @staticmethod
139    def get_structure(atoms, cls=None, **cls_kwargs):
140        """
141        Returns pymatgen structure from ASE Atoms.
142
143        Args:
144            atoms: ASE Atoms object
145            cls: The Structure class to instantiate (defaults to pymatgen structure)
146            **cls_kwargs: Any additional kwargs to pass to the cls
147
148        Returns:
149            Equivalent pymatgen.core.structure.Structure
150        """
151
152        cls = Structure if cls is None else cls
153
154        symbols = atoms.get_chemical_symbols()
155        positions = atoms.get_positions()
156        lattice = atoms.get_cell()
157
158        # Get the site magmoms from the ASE Atoms objects.
159        if getattr(atoms, "calc", None) is not None and getattr(atoms.calc, "results", None) is not None:
160            magmoms = atoms.calc.results.get("magmoms", None)
161            charges = atoms.calc.results.get("charges", None)
162        else:
163            magmoms = None
164            charges = None
165
166        if atoms.has("initial_magmoms"):
167            initial_magmoms = atoms.get_initial_magnetic_moments()
168        else:
169            initial_magmoms = None
170        if atoms.has("initial_charges"):
171            initial_charges = atoms.get_initial_charges()
172        else:
173            initial_charges = None
174        if atoms.has("oxi_states"):
175            oxi_states = atoms.get_array("oxi_states")
176        else:
177            oxi_states = None
178
179        # If the ASE Atoms object has constraints, make sure that they are of the
180        # kind FixAtoms, which are the only ones that can be supported in Pymatgen.
181        # By default, FixAtoms fixes all three (x, y, z) dimensions.
182        if atoms.constraints:
183            unsupported_constraint_type = False
184            constraint_indices = []
185            for constraint in atoms.constraints:
186                if isinstance(constraint, FixAtoms):
187                    constraint_indices.extend(constraint.get_indices().tolist())
188                else:
189                    unsupported_constraint_type = True
190            if unsupported_constraint_type:
191                warnings.warn("Only FixAtoms is supported by Pymatgen. Other constraints will not be set.")
192            sel_dyn = [[False] * 3 if atom.index in constraint_indices else [True] * 3 for atom in atoms]
193        else:
194            sel_dyn = None
195
196        # Return a Molecule object if that was specifically requested;
197        # otherwise return a Structure object as expected
198        if cls == Molecule:
199            structure = cls(symbols, positions, **cls_kwargs)
200        else:
201            structure = cls(lattice, symbols, positions, coords_are_cartesian=True, **cls_kwargs)
202
203        # Set the site magmoms in the Pymatgen structure object
204        # Note: ASE distinguishes between initial and converged
205        # magnetic moment site properties, whereas pymatgen does not. Therefore, we
206        # have to distinguish between these two when constructing the Structure/Molecule.
207        # The mapping selected here is:
208        # ASE initial magmom <--> Pymatgen "magmom"
209        # ASE final magmom <--> Pymatgen "final_magmom"
210        # ASE initial charge <--> Pymatgen "charge"
211        # ASE final charge <--> Pymatgen "final_charge"
212        if magmoms is not None:
213            structure.add_site_property("final_magmom", magmoms)
214        if initial_magmoms is not None:
215            structure.add_site_property("magmom", initial_magmoms)
216        if charges is not None:
217            structure.add_site_property("final_charge", charges)
218        if initial_charges is not None:
219            structure.add_site_property("charge", initial_charges)
220        if sel_dyn is not None and ~np.all(sel_dyn):
221            structure.add_site_property("selective_dynamics", sel_dyn)
222
223        # Add oxidation states by site
224        if oxi_states is not None:
225            structure.add_oxidation_state_by_site(oxi_states)
226
227        # Add any remaining site properties to the Pymatgen structure object
228        for prop in atoms.arrays:
229            if prop not in [
230                "numbers",
231                "positions",
232                "magmom",
233                "final_magmom",
234                "charge",
235                "final_charge",
236                "oxi_states",
237            ]:
238                structure.add_site_property(prop, atoms.get_array(prop).tolist())
239
240        return structure

242    @staticmethod
243    def get_molecule(atoms, cls=None, **cls_kwargs):
244        """
245        Returns pymatgen molecule from ASE Atoms.
246
247        Args:
248            atoms: ASE Atoms object
249            cls: The Molecule class to instantiate (defaults to pymatgen molecule)
250            **cls_kwargs: Any additional kwargs to pass to the cls
251
252        Returns:
253            Equivalent pymatgen.core.structure.Molecule
254        """
255
256        cls = Molecule if cls is None else cls
257        molecule = AseAtomsAdaptor.get_structure(atoms, cls=cls, **cls_kwargs)
258        if atoms.has("initial_charges"):
259            charge = round(np.sum(atoms.get_initial_charges()))
260        else:
261            charge = 0
262        if atoms.has("initial_magmoms"):
263            mult = round(np.sum(atoms.get_initial_magnetic_moments())) + 1
264        else:
265            mult = 1
266        molecule.set_charge_and_spin(charge, spin_multiplicity=mult)
267
268        return molecule
