 25    """
 26    This class behaves exactly like a normal Prefect workflow, where we add some
 27    common utilities and pre-submit tasks. For example, there is the `run_cloud`
 28    method, which allows us to register a calculation to a database table before

 35    """
 36    In many ETL workflows, we may also want the result of the some terminating
 37    task directly. This would save us from having to go find the result in
 38    the database. So by setting `result_task`, we are able get access a specific task's
 39    result -- directly as a python object(s). This is optional.
 40    """

 43    """
 44    The name of the project in the  that this workflow is associated with. This 
 45    attribute is mainly just for organizing workflows in the Prefect Cloud interface.
 46    """

 49    """
 50    The supervised-staged-shell task (or S3Task) that this workflow uses to run.

 53    without storing results in the database.

 57    """

 60    result table! The table should use 
 61    `simmate.database.base_data_types.Calculation`

127    def run_cloud(
128        self,
129        labels: List[str] = [],
130        wait_for_run: bool = True,
131        **kwargs,
132    ) -> str:
133        """
134        This schedules the workflow to run remotely on Prefect Cloud.
135
136        #### Parameters
137
138        - `labels`:
139            a list of labels to schedule the workflow with
140
141        - `wait_for_run`:
142            whether to wait for the workflow to finish. If False, the workflow
143            will simply be submitted and then exit. The default is True.
144
145        - `**kwargs`:
146            all options that are normally passed to the workflow.run() method
147
148        #### Returns
149
150        - The flow run id that was used in prefect cloud.
151
152
153        #### Usage
154
155        Make sure you have Prefect properly configured and have registered your
156        workflow with the backend.
157
158        Note that this method can be viewed as a fork of:
159            - from prefect.tasks.prefect.flow_run import create_flow_run
160        It can also be viewed as a more convenient way to call to client.create_flow_run.
161        I don't accept any other client.create_flow_run() inputs besides 'labels'.
162        This may change in the future if I need to set flow run names or schedules.
163        """
164
165        # Grab the logger as we will print useful information below.
166        # In some cases, there is no logger present, so we just set logger=None.
167        # This possibility is why we use `if logger else None` statements below.
168        logger = getattr(prefect.context, "logger", None)
169
170        # Grab the Flow's ID from Prefect Cloud
171        logger.debug("Looking up flow metadata...") if logger else None
172        flow_view = FlowView.from_flow_name(
173            self.name,
174            project_name=self.project_name,
175        )
176
177        # Serialize all of the parameters so they can be properly submitted
178        parameters_serialized = self._serialize_parameters(kwargs)
179
180        # Now we submit the workflow
181        logger.info(f"Creating flow run for {self.name}...") if logger else None
182        client = Client()
183        flow_run_id = client.create_flow_run(
184            flow_id=flow_view.flow_id,
185            parameters=parameters_serialized,
186            labels=labels,
187        )
188
189        # we log the website url to the flow for the user
190        run_url = client.get_cloud_url("flow-run", flow_run_id, as_user=False)
191        logger.info(f"Created flow run: {run_url}") if logger else None
192
193        # -----------------------------------
194        # This part is unique to Simmate. Because we often want to save some info
195        # to our database even before the calculation starts/finishes, we do that
196        # here. An example is storing the structure and prefect id that we just
197        # submitted.
198        # BUG: Will there be a race condition here? What if the workflow finishes
199        # and tries writing to the databse before this is done?
200        register_kwargs = {
201            key: kwargs[key] for key in kwargs if key in self.register_kwargs
202        }
203        calc = self._register_calculation(
204            flow_run_id,
205            **register_kwargs,
206        )  # !!! should I return the calc to the user?
207        # -----------------------------------
208
209        # if we want to wait until the job is complete, we do that here
210        if wait_for_run:
211            flow_run_view = self.wait_for_flow_run(flow_run_id)
212            # return flow_run_view # !!! Should I return this instead?
213        # return the flow_run_id for the user
214        return flow_run_id

216    def _register_calculation(self, flow_run_id: str, **kwargs):
217        """
218        If the workflow is linked to a calculation table in the Simmate database,
219        this adds the flow run to the Simmate database.
220
221        This method should not be called directly as it is used within the
222        run_cloud() method.
223        """
224        # If there's no calculation database table in Simmate for this workflow,
225        # just skip this step. Otherwise save/load the calculation to our table
226        if self.calculation_table:
227            calculation = self.calculation_table.from_prefect_id(
228                id=flow_run_id,
229                **kwargs,
230            )
231            return calculation

233    def wait_for_flow_run(self, flow_run_id: str):
234        """
235        Waits for a given flow run to complete
236
237        This method is a direct fork of...
238            from prefect.tasks.prefect.flow_run import wait_for_flow_run
239
240        It does exactly the same thing where I just assume I want to stream logs.
241        """
242
243        flow_run = FlowRunView.from_flow_run_id(flow_run_id)
244
245        for log in flow_run_module.watch_flow_run(flow_run_id):
246            message = f"Flow {flow_run.name}: {log.message}"
247            prefect.context.logger.log(log.level, message)
248        # Return the final view of the flow run
249        return flow_run.get_latest()

251    @staticmethod
252    def _serialize_parameters(parameters: dict) -> dict:
253        """
254        Converts input parameters to json-sealiziable objects that Prefect can
255        use.
256
257        This method should not be called directly as it is used within the
258        run_cloud() method.
259        """
260        # Because many flows allow object-type inputs (such as structure object),
261        # we need to serialize these inputs before scheduling them with prefect
262        # cloud. To do this, I only check for two potential methods:
263        #     as_dict
264        #     to_dict
265        # These were chosen based on common pymatgen methods, but I may change
266        # this in the future. As another alternative, I could also cloudpickle
267        # input objects when they are not JSON serializable.
268        # OPTIMIZE: Prefect current tries to JSON serialize within their
269        # client.create_flow_run method. In the future, we may want to move
270        # this functionality there.
271
272        parameters_serialized = {}
273        for parameter_key, parameter_value in parameters.items():
274
275            try:
276                json.dumps(parameter_value)
277            except TypeError:
278                if hasattr(parameter_value, "as_dict"):
279                    parameter_value = parameter_value.as_dict()
280                elif hasattr(parameter_value, "to_dict"):
281                    parameter_value = parameter_value.to_dict()
282                else:
283                    parameter_value = cloudpickle.dumps(parameter_value)
284            parameters_serialized[parameter_key] = parameter_value
285        return parameters_serialized
